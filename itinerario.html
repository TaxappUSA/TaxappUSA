async function updateQueueTimes() { const now = new Date(); const hour = now.getHours(); const city = document.getElementById('citySelector').value; if (city !== 'orlando') return; const parks = citiesConfig.orlando.filter(p => p.parkId); for (const park of parks) { const el = document.getElementById(park.id); if (!el) continue; try { const proxy = "https://corsproxy.io/?"; const wikiUrl = `https://api.themeparks.wiki/v1/entity/${park.parkId}/live`; const response = await fetch(proxy + encodeURIComponent(wikiUrl)); const data = await response.json(); let average = 0; let activeRides = 0; if (data.liveData) { // Filtramos solo atracciones que estén ABIERTAS y tengan tiempo de espera real const waits = data.liveData.filter(a => a.status === "OPERATING" && a.queue?.STANDBY?.waitTime !== null && a.entityType === "ATTRACTION" ).map(a => a.queue.STANDBY.waitTime); activeRides = waits.length; average = activeRides ? waits.reduce((a, b) => a + b, 0) / activeRides : 0; } // Lógica de visualización fiel al forecast if (activeRides === 0 && (hour < 8 || hour > 22)) { el.innerText = "CERRADO 🌙"; el.className = "queue-status q-off"; } else if (activeRides === 0) { el.innerText = "SIN INFO 🔄"; el.className = "queue-status q-off"; } else { // Ajuste de umbrales para que "Quiet" (Tranquilo) se refleje correctamente if (average <= 20) { el.innerText = "BAJA 🟢"; el.className = "queue-status q-low"; } else if (average <= 45) { el.innerText = "MED 🟡"; el.className = "queue-status q-med"; } else { el.innerText = "ALTA 🔴"; el.className = "queue-status q-high"; } } } catch (e) { console.error("Error en " + park.name, e); el.innerText = "Error 🔄"; el.className = "queue-status q-off"; } } }
